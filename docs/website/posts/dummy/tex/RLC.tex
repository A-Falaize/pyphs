\section{Constitutive relations}
%

%
\\
%
Dissipation function $ \mathbf{z}(\mathbf{w}) = \left(\begin{array}{c}\mathrm{R} \cdot w_{\mathrm{R}}\end{array}\right) ; $ 
%
\\
